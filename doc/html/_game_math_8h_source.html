146  void rotate(float angle, const Vector3f& axis) {
147  float sinHalfAngle = sinf(toRadian(angle/2));
148  float cosHalfAngle = cosf(toRadian(angle/2));
149 
150  float Rx = axis.x * sinHalfAngle;
151  float Ry = axis.y * sinHalfAngle;
152  float Rz = axis.z * sinHalfAngle;
153  float Rw = cosHalfAngle;
154  Quaternion rotationQ(Rx, Ry, Rz, Rw);
155 
156  Quaternion conjugateQ = rotationQ.conjugate();
157  Quaternion W = this->multQuaternion(rotationQ).mult(conjugateQ);
158 
159  x = W.x;
160  y = W.y;
161  z = W.z;
162  }

164  Quaternion multQuaternion(const Quaternion& q) {

168  float _z = (q.w * z) + (q.x * y) - (q.y * x);

271  void setPerspectiveProjection(float fov, float width, float height, float zNear, float zFar) {
272  float ar = width / height;

274  float tanHalfFOV = tanf(toRadian(fov / 2.0f));
275

277  m[1][0] = 0.0f; m[1][1] = 1.0f / tanHalfFOV; m[1][2] = 0.0f; m[1][3] = 0.0;
278  m[2][0] = 0.0f; m[2][1] = 0.0f; m[2][2] = (-zNear - zFar) / zRange; m[2][3] = 2.0f * zFar * zNear / zRange;
279  m[3][0] = 0.0f; m[3][1] = 0.0f; m[3][2] = 1.0f; m[3][3] = 0.0;
